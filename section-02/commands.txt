kubectl get all

kubectl get <contenttype>
kubectl logs <podname>

kubectl expose pod <podname> --port=6379 --name <servicename> --dry-run -o yaml


#options
--dry-run  	will not create the resources only test.
-o yaml		format output yaml|json
